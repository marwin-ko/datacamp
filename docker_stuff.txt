
#
# How to upgrade RStudio (go to https://rstudio.com/products/rstudio/download/ and copy url from download link)
#
RUN cd /tmp && wget https://download2.rstudio.org/server/bionic/amd64/rstudio-server-1.3.1093-amd64.deb && gdebi -n rstudio-server-1.3.1093-amd64.deb && \
ln -s /usr/lib/rstudio-server /usr/local/lib/rstudio-server && \
chown -R root:root /usr/lib/rstudio-server
